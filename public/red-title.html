Verilog RTL 核心实现

module tod_counter (
    input  wire        clk_i,
    input  wire        rst_n_i,
    output reg  [31:0] tod_ns_o,
    output reg  [31:0] tod_us_o,
    output reg  [31:0] tod_ms_o,
    output reg  [31:0] tod_sec_o
);

    wire carry_ns, carry_us, carry_ms;

    // 纳秒计数：假设 clk = 250MHz → 周期 = 4ns
    always @(posedge clk_i or negedge rst_n_i) begin
        if (!rst_n_i)
            tod_ns_o <= 32'd0;
        else if (carry_ns)
            tod_ns_o <= 32'd0;
        else
            tod_ns_o <= tod_ns_o + 32'd4;
    end
    assign carry_ns = (tod_ns_o >= 32'd1000);

    // 微秒计数
    always @(posedge clk_i or negedge rst_n_i) begin
        if (!rst_n_i)
            tod_us_o <= 32'd0;
        else if (carry_ns)
            tod_us_o <= tod_us_o + 32'd1;
    end

    // 毫秒计数
    assign carry_us = (tod_us_o == 32'd999) && carry_ns;
    always @(posedge clk_i or negedge rst_n_i) begin
        if (!rst_n_i)
            tod_ms_o <= 32'd0;
        else if (carry_us)
            tod_ms_o <= tod_ms_o + 32'd1;
    end

    // 秒计数
    assign carry_ms = (tod_ms_o == 32'd999) && carry_us;
    always @(posedge clk_i or negedge rst_n_i) begin
        if (!rst_n_i)
            tod_sec_o <= 32'd0;
        else if (carry_ms)
            tod_sec_o <= tod_sec_o + 32'd1;
    end

endmodule
